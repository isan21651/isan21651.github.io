  <script>
    let balance = 5000;
    const balanceEl = document.getElementById('balance');
    const historyEl = document.getElementById('history');

    function deposit() {
      let amount = prompt("Enter deposit amount:");
      if (amount && !isNaN(amount)) {
        balance += Number(amount);
        updateBalance(`+${amount}‡ß≥ (Deposit)`);
      }
    }

    function withdraw() {
      let amount = prompt("Enter withdraw amount:");
      if (amount && !isNaN(amount)) {
        if (Number(amount) <= balance) {
          balance -= Number(amount);
          updateBalance(`-${amount}‡ß≥ (Withdraw)`);
        } else {
          alert("Insufficient balance!");
        }
      }
    }

    function send() {
      let name = prompt("Enter recipient name:");
      let amount = prompt("Enter amount to send:");
      if (name && amount && !isNaN(amount)) {
        if (Number(amount) <= balance) {
          balance -= Number(amount);
          updateBalance(`-${amount}‡ß≥ Sent to ${name}`);
        } else {
          alert("Not enough balance!");
        }
      }
    }

    function updateBalance(text) {
      balanceEl.textContent = balance;
      historyEl.innerHTML += `<br>${text}`;
    }
